typescript// lib/grok-client.ts
export class GrokClient {
  private apiKey: string;
  
  constructor() {
    this.apiKey = process.env.GROK_API_KEY!;
  }
  
  async generateChapter(prompt: string): Promise<string> {
    const response = await fetch('https://api.x.ai/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.apiKey}`
      },
      body: JSON.stringify({
        model: 'grok-2-1212', // Latest model
        messages: [
          { 
            role: 'system', 
            content: 'You are a romance novelist...' 
          },
          { 
            role: 'user', 
            content: prompt 
          }
        ],
        temperature: 0.9, // High creativity
        max_tokens: 4096 // ~3k words
      })
    });
    
    const data = await response.json();
    return data.choices[0].message.content;
  }
}

typescript// lib/story-generator.ts
import { GrokClient } from './grok-client';
import { authorProfiles } from './data/authors';
import { beatStructures } from './data/beats';

export class StoryGenerator {
  private grok = new GrokClient();
  
  async generateStory(config: StoryConfig): Promise<Chapter[]> {
    // 1. Plan foreshadowing upfront
    const foreshadowingPlan = this.planForeshadowing(
      config.totalChapters
    );
    
    const chapters: Chapter[] = [];
    
    // 2. Generate each chapter sequentially
    for (let i = 1; i <= config.totalChapters; i++) {
      const chapter = await this.generateChapter(
        config,
        i,
        chapters,
        foreshadowingPlan
      );
      chapters.push(chapter);
      
      // Save progress (for jail friend batch generation)
      await this.saveChapter(config.storyId, chapter);
    }
    
    return chapters;
  }
  
  private planForeshadowing(totalChapters: number): ForeshadowingDevice[] {
    // Simple formula: plant early, remind middle, payoff late
    const devices = [
      'mysterious_scar',
      'forbidden_artifact', 
      'prophetic_dream',
      'hidden_identity'
    ];
    
    return devices.map((name, i) => ({
      name,
      plantChapter: Math.floor(totalChapters * 0.15) + i,
      remindChapters: [
        Math.floor(totalChapters * 0.4),
        Math.floor(totalChapters * 0.65)
      ],
      payoffChapter: Math.floor(totalChapters * 0.85) + i
    }));
  }
  
  private async generateChapter(
    config: StoryConfig,
    chapterNum: number,
    previousChapters: Chapter[],
    foreshadowingPlan: ForeshadowingDevice[]
  ): Promise<Chapter> {
    // Build context from previous chapters
    const context = previousChapters
      .slice(-3) // Last 3 chapters for context
      .map(c => c.summary)
      .join('\n\n');
    
    // Check what foreshadowing is active for this chapter
    const toPlant = foreshadowingPlan.filter(
      f => f.plantChapter === chapterNum
    );
    const toRemind = foreshadowingPlan.filter(
      f => f.remindChapters.includes(chapterNum)
    );
    const toPayoff = foreshadowingPlan.filter(
      f => f.payoffChapter === chapterNum
    );
    
    // Build prompt with foreshadowing instructions
    const prompt = this.buildPrompt({
      config,
      chapterNum,
      context,
      toPlant,
      toRemind,
      toPayoff
    });
    
    const content = await this.grok.generateChapter(prompt);
    
    // Extract summary for future context
    const summary = await this.summarizeChapter(content);
    
    return {
      number: chapterNum,
      content,
      summary,
      foreshadowingPlanted: toPlant.map(f => f.name),
      foreshadowingPaid: toPayoff.map(f => f.name)
    };
  }
  
  private buildPrompt(params: PromptParams): string {
    return `
You are writing Chapter ${params.chapterNum} of ${params.config.totalChapters}.

AUTHOR VOICE: Blend of ${params.config.authorBlend.join(', ')}
SPICE LEVEL: ${params.config.spiceLevel}/5
CURRENT BEAT: ${this.getCurrentBeat(params.chapterNum, params.config)}

${params.toPlant.length > 0 ? `
PLANT THESE DETAILS (subtle, not obvious):
${params.toPlant.map(f => `- ${f.name}`).join('\n')}
` : ''}

${params.toRemind.length > 0 ? `
CALLBACK TO THESE EARLIER DETAILS:
${params.toRemind.map(f => `- ${f.name}`).join('\n')}
` : ''}

${params.toPayoff.length > 0 ? `
RESOLVE THESE PLOT THREADS:
${params.toPayoff.map(f => `- ${f.name}`).join('\n')}
` : ''}

PREVIOUS CONTEXT:
${params.context}

Write ~3000 words. Follow prose quality rules:
- SHOW don't TELL (no "felt", "realized", "knew")
- Concrete sensory details
- Spice level ${params.config.spiceLevel} vocabulary

Write the chapter now.
`;
  }
}
